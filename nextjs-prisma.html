app/actions.ts 
// app/actions.ts
"use server"  // ← บอก Next.js ว่านี่คือ Server-side code

import { prisma } from "@/lib/prisma"
import { revalidatePath } from "next/cache"

// ✅ CREATE — เพิ่ม Project ใหม่
export async function createProject(formData: FormData) {
  const name = formData.get("name") as string
  const description = formData.get("description") as string

  await prisma.project.create({
    data: { name, description }
  })

  revalidatePath("/")
}

// ✏️ UPDATE — แก้ไข Project
export async function updateProject(formData: FormData) {
  const id = formData.get("id") as string
  const name = formData.get("name") as string
  const description = formData.get("description") as string

  await prisma.project.update({
    where: { id },          // ← หา record ที่ต้องการแก้
    data: { name, description }  // ← ข้อมูลใหม่
  })

  revalidatePath("/")
}

// 🗑️ DELETE — ลบ Project
export async function deleteProject(formData: FormData) {
  const id = formData.get("id") as string

  await prisma.project.delete({
    where: { id }  // ← ลบตาม id
  })

  revalidatePath("/")
}

📊 สรุป CRUD ทั้ง 4 ตัว

Operation Prisma Method ใช้ยังไง
Create prisma.project.create() ส่ง data object ที่ต้องการสร้าง
Read prisma.project.findMany() ดึงทั้งหมด หรือ findUnique() ดึงตัวเดียว
Update prisma.project.update() ระบุ where + data ที่จะแก้
Delete prisma.project.delete() ระบุ where → ลบทิ้ง

💡 เทคนิค: ส่ง id ผ่าน hidden input

ทั้ง Update และ Delete ต้องรู้ว่าจะแก้/ลบ ตัวไหน — เราใส่ <input type="hidden" name="id" value={p.id} /> ไว้ในฟอร์ม เพื่อส่ง id ไปกับ FormData โดยที่ User ไม่เห็น

🔑 Key Insight — ทำไม Server Actions ถึงเจ๋ง?

  • ✓ ไม่ต้องสร้าง API Route แยก — เขียนฟังก์ชันเดียวจบ
  • ✓ "use server" ทำให้ Next.js รู้ว่าต้องรันบน Server
  • ✓ ผูก form action={fn} ได้ตรงๆ — ไม่ต้อง onSubmit + fetch
  • ✓ Server Component ดึงข้อมูลได้เลย ไม่ต้อง useState/useEffect

prisma/schema.prisma 
// prisma/schema.prisma

generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "postgresql"  // หรือ "sqlite" สำหรับลองในเครื่อง
  url      = env("DATABASE_URL")
}

// 📂 Model: Project (โปรเจกต์หลัก)
model Project {
  id          String   @id @default(cuid())
  name        String          // ชื่อโปรเจกต์
  description String?         // รายละเอียด (optional)
  status      ProjectStatus @default(ACTIVE)
  createdAt   DateTime @default(now())
  updatedAt   DateTime @updatedAt

  // 🔗 ความสัมพันธ์: 1 Project มีหลาย Task
  tasks       Task[]
}

// ✅ Model: Task (งานย่อย)
model Task {
  id          String     @id @default(cuid())
  title       String
  completed   Boolean    @default(false)
  priority    Priority   @default(MEDIUM)
  createdAt   DateTime   @default(now())

  // 🔗 FK: แต่ละ Task ต้องอยู่ใน Project
  projectId   String
  project     Project    @relation(fields: [projectId], references: [id])
}

// 🎨 Enums
enum ProjectStatus {
  ACTIVE
  COMPLETED
  ARCHIVED
}

enum Priority {
  LOW
  MEDIUM
  HIGH
}

Phase 3 Special Topic

Data Validation with Zod

ทำไมต้อง Validate? เพราะ User ส่งอะไรมาก็ได้ — เราต้องป้องกัน DB ของเรา!

🤔 ทำไมต้อง Validate ก่อนเข้า Database?

❌ ไม่ Validate
  • • User ส่งชื่อว่าง → Error จาก DB
  • • User ส่งอีเมลผิด format → ข้อมูลเสีย
  • • User ส่ง string 10,000 ตัว → DB บวม
  • • Hacker ส่ง SQL Injection → 💀
✅ ใช้ Zod Validate
  • • ดักข้อมูลก่อนถึง DB — ปลอดภัย
  • • Error Message ชัดเจน → UX ดี
  • • Type-safe ทั้ง Runtime + Compile time
  • • Schema เดียวใช้ได้ทั้ง Server & Client

💡 คิดง่ายๆ: TypeScript ดักข้อผิดพลาดตอนเขียนโค้ด (compile time) แต่ Zod ดักข้อผิดพลาดตอนรับข้อมูลจริงจาก User (runtime) — ต้องมีทั้งคู่!

// lib/validations.ts
import { z } from "zod"

// 📝 Schema สำหรับสร้าง Project
export const createProjectSchema = z.object({
  // ✅ ชื่อโปรเจกต์ต้องมี 5-50 ตัวอักษร
  name: z.string()
    .min(5, { message: "ชื่อโปรเจกต์ต้องมีอย่างน้อย 5 ตัวอักษร" })
    .max(50, { message: "ชื่อโปรเจกต์ต้องไม่เกิน 50 ตัวอักษร" }),

  // ✅ รายละเอียด (ไม่บังคับ แต่ถ้ากรอกต้องไม่เกิน 500 ตัว)
  description: z.string()
    .max(500, { message: "รายละเอียดต้องไม่เกิน 500 ตัวอักษร" })
    .optional(),

  // ✅ อีเมลต้องถูก Format
  ownerEmail: z.string()
    .email({ message: "รูปแบบอีเมลไม่ถูกต้อง" }),
})

// 🎯 ดึง Type จาก Schema (ไม่ต้องเขียน interface ซ้ำ!)
export type CreateProjectInput = z.infer<typeof createProjectSchema>

// 📝 Schema สำหรับสร้าง Task
export const createTaskSchema = z.object({
  title: z.string()
    .min(3, { message: "ชื่อ Task ต้องมีอย่างน้อย 3 ตัวอักษร" })
    .max(100, { message: "ชื่อ Task ต้องไม่เกิน 100 ตัวอักษร" }),
  projectId: z.string()
    .cuid({ message: "Project ID ไม่ถูกต้อง" }),
  priority: z.enum(["LOW", "MEDIUM", "HIGH"])
    .default("MEDIUM"),
})

Method ใช้กับ ตัวอย่าง
.min(n) string, number z.string().min(5)
.max(n) string, number z.string().max(50)
.email() string z.string().email()
.optional() ทุก type z.string().optional()
.enum() string z.enum(["A","B"])

app/actions.ts — Pro Version 
"use server"

import { prisma } from "@/lib/prisma"
import { revalidatePath } from "next/cache"
import type { ActionResult } from "@/lib/types"

export async function createProject(
  formData: FormData
): Promise<ActionResult> {
  try {
    const name = formData.get("name") as string
    const description = formData.get("description") as string

    // 1️⃣ Validate (จะสอนเพิ่มใน Phase 6!)
    if (!name || name.length < 5) {
      return {
        success: false,
        message: "ชื่อโปรเจกต์ต้องมีอย่างน้อย 5 ตัวอักษร",
        errors: { name: ["ต้องมีอย่างน้อย 5 ตัวอักษร"] }
      }
    }

    // 2️⃣ บันทึกลง DB
    await prisma.project.create({
      data: { name, description }
    })

    // 3️⃣ Refresh ข้อมูลบนหน้าเว็บ
    revalidatePath("/projects")

    // 4️⃣ ส่งสถานะสำเร็จกลับ
    return {
      success: true,
      message: "สร้างโปรเจกต์สำเร็จ! 🎉"
    }

  } catch (error) {
    // 5️⃣ ดัก Error ที่ไม่คาดคิด (DB ล่ม, Network error ฯลฯ)
    console.error("Create project failed:", error)
    return {
      success: false,
      message: "เกิดข้อผิดพลาด กรุณาลองใหม่อีกครั้ง"
    }
  }
}

import { useActionState } from "react"

}

"use client"

import { useOptimistic } from "react"
import { toggleTask } from "../actions"

type Task = { id: string; title: string; completed: boolean }

export default function TaskList({ tasks }: { tasks: Task[] }) {
  // ✅ useOptimistic — สร้าง "state ชั่วคราว" ที่เปลี่ยนทันที
  const [optimisticTasks, addOptimistic] = useOptimistic(
    tasks,
    // ฟังก์ชันนี้จะถูกเรียกเมื่อ addOptimistic() ทำงาน
    (currentTasks: Task[], taskId: string) =>
      currentTasks.map((t) =>
        t.id === taskId ? { ...t, completed: !t.completed } : t
      )
  )

  async function handleToggle(taskId: string) {
    // 1️⃣ เปลี่ยน UI ทันที (ไม่รอ Server)
    addOptimistic(taskId)
    // 2️⃣ ส่งคำสั่งไป Server (ทำพร้อมกัน)
    await toggleTask(taskId)
  }

  return (
    <ul>
      {optimisticTasks.map((task) => (
        <li key={task.id} className="flex items-center gap-2">
          <input
            type="checkbox"
            checked={task.completed}
            onChange={() => handleToggle(task.id)}
          />
          <span className={task.completed ? "line-through" : ""}>
            {task.title}
          </span>
        </li>
      ))}
    </ul>
  )
}

// lib/validations.ts
import { z } from "zod"

// 📝 Zod Schema = "กฎ" ของข้อมูล
export const createProjectSchema = z.object({
  name: z.string()
    .min(5, { message: "ชื่อต้องมีอย่างน้อย 5 ตัวอักษร" })
    .max(50, { message: "ชื่อต้องไม่เกิน 50 ตัวอักษร" }),
  description: z.string()
    .max(500)
    .optional(),
  ownerEmail: z.string()
    .email({ message: "รูปแบบอีเมลไม่ถูกต้อง" }),
})

// ─────────────────────────────────────────────
// ⭐ z.infer — ดึง TypeScript Type จาก Zod Schema
// ─────────────────────────────────────────────
// ไม่ต้องเขียน interface ซ้ำ! Zod สร้างให้อัตโนมัติ:
export type CreateProjectInput = z.infer<typeof createProjectSchema>

// ☝️ CreateProjectInput จะเป็น:
// {
//   name: string
//   description?: string | undefined
//   ownerEmail: string
// }
// → Zod ดูจาก Schema แล้วสร้าง Type ให้เอง! 🔥

⭐ app/actions.ts — Ultimate Version 
"use server"

import { prisma } from "@/lib/prisma"
import { revalidatePath } from "next/cache"
import { createProjectSchema } from "@/lib/validations"
import type { ActionResult } from "@/lib/types"

// ============================================
// ⭐ CREATE PROJECT — Zod + Prisma + Error Handling
// ============================================
export async function createProject(
  _prevState: ActionResult,  // state ก่อนหน้า (จาก useActionState)
  formData: FormData
): Promise<ActionResult> {
  try {
    // ──────────────────────────────────
    // STEP 1: ดึงค่าจากฟอร์ม
    // ──────────────────────────────────
    const rawData = {
      name: formData.get("name"),
      description: formData.get("description"),
      ownerEmail: formData.get("ownerEmail"),
    }

    // ──────────────────────────────────
    // STEP 2: ✅ Validate ด้วย Zod
    // ──────────────────────────────────
    const validated = createProjectSchema.safeParse(rawData)

    if (!validated.success) {
      // ❌ ข้อมูลไม่ผ่าน → ส่ง Error กลับให้ฟอร์ม
      return {
        success: false,
        message: "กรุณาตรวจสอบข้อมูลที่กรอก",
        errors: validated.error.flatten().fieldErrors as
          Record<string, string[]>,
      }
    }

    // ──────────────────────────────────
    // STEP 3: 💾 บันทึกลง Database
    // ──────────────────────────────────
    await prisma.project.create({
      data: {
        name: validated.data.name,
        description: validated.data.description,
        // ownerEmail สามารถเก็บตาม Business Logic
      },
    })

    // ──────────────────────────────────
    // STEP 4: 🔄 Refresh หน้าเว็บ
    // ──────────────────────────────────
    revalidatePath("/projects")

    // ──────────────────────────────────
    // STEP 5: ✅ ส่งผลลัพธ์สำเร็จ
    // ──────────────────────────────────
    return {
      success: true,
      message: "สร้างโปรเจกต์สำเร็จ! 🎉",
    }

  } catch (error) {
    // ──────────────────────────────────
    // 🛑 Unexpected Error (DB ล่ม ฯลฯ)
    // ──────────────────────────────────
    console.error("[createProject] Error:", error)
    return {
      success: false,
      message: "เกิดข้อผิดพลาดภายในระบบ กรุณาลองใหม่",
    }
  }
}
